vim-8 or later built-in plugin or package management directories:

~/.vim/pack/vendor/start/nerdtree,
~/.vim/pack/vendor/opt/foobar,

1. "vendor" is just a name, it can be named otherwise like: plugins, plugin_dir, etc.;
2. plugin under start is automatically loaded;
3. plugin under opt needs to be loaded in ~/.vimrc: " packadd! foobar ";


// vim-lsp
$ git clone --depth=1 https://github.com/prabirshrestha/vim-lsp.git ~/.vim/pack/vendor/opt/vim-lsp/
$ vim -u NONE -c "helptags ~/.vim/pack/vendor/opt/vim-lsp/doc" -c q

// vim-lsp-settings
$ git clone --depth=1 https://github.com/prabirshrestha/vim-lsp-settings.git ~/.vim/pack/vendor/opt/vim-lsp-settings/
$ vim -u NONE -c "helptags ~/.vim/pack/vendor/opt/vim-lsp-settings/doc" -c q

$ vi ~/.vimrc
" vim-lsp, vim-lsp-settings,
" similar to neovim: tagfunc is set to vim.lsp.tagfunc(),
" add vim-lsp in vim only install in `opt` (not `start`) directory then packadd,
" neovim has built-in lsp, and should source a modified copy eg. ~/.vimrc2,
packadd! vim-lsp          " comment in neovim
packadd! vim-lsp-settings " comment in neovim
set tagfunc=lsp#tagfunc   " comment in neovim
$

// directory tree, use built-in netrw instead of nerdtree
$ git clone --depth=1 https://github.com/preservim/nerdtree.git ~/.vim/pack/vendor/start/nerdtree/
$ vim -u NONE -c "helptags ~/.vim/pack/vendor/start/nerdtree/doc" -c q

// symbol outline
$ git clone --depth=1 https://github.com/preservim/tagbar.git ~/.vim/pack/vendor/start/tagbar/
$ vim -u NONE -c "helptags ~/.vim/pack/vendor/start/tagbar/doc" -c q

// automatic ctags manager
$ git clone --depth=1 https://github.com/ludovicchabant/vim-gutentags.git ~/.vim/pack/vendor/start/gutentags/
$ vim -u NONE -c "helptags ~/.vim/pack/vendor/start/gutentags/doc" -c q

// jump to definition
$ git clone --depth=1 https://github.com/universal-ctags/ctags.git
$ cd ctags && bash autogen.sh && ./configure && make && sudo make install
$
